### Usage
#### Example
```
package main

import (
	"fmt"
	"github.com/hslam/rbtree"
)

func main() {
	t := rbtree.New()
	str := item("Hello World")
	t.Insert(str)
	fmt.Println(t.Search(str))
	t.Delete(str)
}

type item string

func (n item) Less(b rbtree.Item) bool {
	return n < b.(item)
}
```

#### Output
```
Hello World
```
